- 使用 styled-components

  - 全局样式使用
    - 1. 用 createGlobalStyle 定义全局样式

  ```typescript
    import { createGlobalStyle } from 'styled-components'
    export const Globalstyle = createGlobalStyle`
      body{
      margin:0;
      padding:0;
      font-family:sans-serif;
      background:green;
      }
      `
      // 2.在项目主文件导入样式
      import { Globalstyle } from './style';
      // 3.以样式组件的方式当作标签引入
      import { Globalstyle } from './style';
      render() {
      　　return (
      　　　　<div>
      　　　　　　<Globalstyle/>
      　　　　</div>
      );
  ```
